#include "bits/stdc++.h"
using namespace std;
int nodes, edges;
vector <vector<int>> v;
vector <int> visited, indegree;
void dfs(int i){
    visited[i] = 1;
    cout << i << " ";
    for (auto x : v[i]){
        indegree[x]--;
        if (visited[x] == 0 and indegree[x] == 0)
            dfs(x);
    }
}
int main(){
    cout << "Enter the number of nodes : ";cin >> nodes;
    cout << "Enter the number of edges : ";cin >> edges;
    v.resize(nodes);
    visited.resize(nodes); indegree.resize(nodes);
    for (int i = 0; i < edges; i++){
        cout << "   ";
        int j,k; cin >> j >> k;
        v[j].push_back(k);
        indegree[k] ++;
    }
    cout << "Topological Sort : ";
    for (int i = 0; i < nodes; i++)
        if (visited[i] == 0 and indegree[i] == 0)
            dfs(i);
    return 0;
}
